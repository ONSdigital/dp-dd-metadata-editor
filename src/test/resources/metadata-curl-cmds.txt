
# Get all Dataset Metadata.

    curl \
    -H "Content-Type: application/json" \
    -X GET http://localhost:23000/metadatas | jq "."

# Get a specific Dataset Metadata (by datasetId).

    curl \
    -H "Content-Type: application/json" \
    -X GET http://localhost:23000/metadata/063085a6-6d88-478b-b5e4-8144dc37c57b | jq "."


# Update a Dataset Metadata

    curl \
    -H "Content-Type: application/json" \
    -i \
    -X PUT \
    -d '{
      "jsonMetadata": "{\"Name\":\"Dataset Metadata Test 1\", \"uploadDate\":\"01/01/2017\"}",
      "datasetId": "063085a6-6d88-478b-b5e4-8144dc37c57b",
      "dataResource": "ABCD1234",
      "majorVersion": "1",
      "minorVersion": "1",
      "revisionNotes": "N/a",
      "revisionReason": "Initial update."
    }' \
    http://localhost:23000/metadata/063085a6-6d88-478b-b5e4-8144dc37c57b
